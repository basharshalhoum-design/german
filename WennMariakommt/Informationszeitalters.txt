  <script>
    const voiceSelect = document.getElementById('voiceSelect');
    const stopBtn = document.getElementById('stopBtn');
    const testBtn = document.getElementById('testBtn');
    let voices = [];

    function loadVoices() {
      voices = speechSynthesis.getVoices();
      voiceSelect.innerHTML = '';
      voices.forEach((v, i) => {
        const opt = document.createElement('option');
        opt.value = i;
        opt.textContent = `${v.name} (${v.lang})`;
        voiceSelect.appendChild(opt);
      });
      const deIndex = voices.findIndex(v => /^de/.test(v.lang));
      if (deIndex >= 0) voiceSelect.value = deIndex;
    }

    speechSynthesis.onvoiceschanged = loadVoices;
    loadVoices();

    function speak(text) {
      speechSynthesis.cancel();
      const u = new SpeechSynthesisUtterance(text);
      const v = voices[voiceSelect.value];
      if (v) u.voice = v;
      u.lang = v ? v.lang : 'de-DE';
      u.rate = 1.0;
      u.pitch = 1.0;
      speechSynthesis.speak(u);
    }

    document.querySelectorAll('td.german').forEach(td => {
      td.addEventListener('click', () => speak(td.textContent.trim()));
    });

    stopBtn.addEventListener('click', () => speechSynthesis.cancel());
    testBtn.addEventListener('click', () => speak('Dies ist ein Audiotest auf Deutsch.'));
  </script>
